\section{Robust and Detailed Algorithm Used in Implementation}
\label{app:algorithm}

\vspace{-3mm}

\begin{algorithm}[H]
\caption{Auto-scheduling algorithm: NN ordering with constraint handling}
\begin{algorithmic}[1]
\STATE Filter enabled places from input; assign stable unique ids if missing.
\STATE Build the day list from the input date range; if \texttt{targetDays} is provided, trim the day list to the first \texttt{targetDays} days.
\STATE Partition places into fixed-day buckets (preferredDay) and an unassigned pool.

\STATE Compute travel duration matrix $M$ if coordinates exist and routing mode supports matrix API.
\IF{\texttt{transport} = \texttt{transit}}
  \STATE Skip matrix computation; transit legs will use per-leg routing calls instead of $M$.
\ENDIF

\STATE Order places: if any strict ordering exists, sort by strict order field; else apply nearest-neighbor heuristic using $M$ (or haversine distance if $M$ unavailable).
\IF{\texttt{orderingMode} = \texttt{strict}}
  \STATE Treat strict order field as exact positions in the ordered list.
\ELSIF{\texttt{orderingMode} = \texttt{relative}}
  \STATE Preserve original input order; reorder only places with a strict order field within that stable order.
\ENDIF
\STATE Apply the ordering procedure separately within each preferredDay bucket and within the unassigned pool.
\IF{$M$ unavailable \AND coordinates missing}
  \STATE Keep original input order as a fallback (no distance-based reordering possible).
\ENDIF

\STATE Compute focus preference offset and sparse gap; ignore focus preference offset when compactness = sparse.

\FOR{each day $d$ in date range}
  \STATE Initialise cursor time $t$ based on daily active hours start and focus preference offset.
  \STATE Initialise today's ordered list by concatenating: (i) preferredDay bucket for $d$ (if any), then (ii) remaining unassigned places in their order.
  \WHILE{unscheduled places remain \AND $t$ within day's active window}
    \STATE Retrieve next place $p$ from ordered list.
    \STATE Compute travel time $\tau$ from previous stop to $p$ (using routing API or fallback).
    \IF{\texttt{transport} = \texttt{transit}}
      \STATE Compute $\tau$ (and any per-leg mode details) via a per-leg routing call; use the returned transit/walk steps when available.
    \ELSE
      \STATE Compute $\tau$ from $M$ when available; otherwise fall back to a distance-based estimate (e.g., haversine).
    \ENDIF
    \STATE Propose arrival time: $a = t_{\text{prev\_depart}} + \text{break\_minutes} + \tau$.
    \STATE Apply desired window constraints (soft) and opening-hours constraints (hard) to $a$.
    \IF{$p$ is marked fixedTime}
      \STATE Treat desired window constraints for $p$ as hard (must be satisfied).
    \ENDIF
    \STATE If opening-hours indicate $p$ is closed at $a$, nudge $a$ to the next feasible open slot when possible; otherwise mark as infeasible.
    \IF{resulting interval $(a, a + \text{dwell})$ fits within day's active window}
      \STATE Commit stop $p$ to day $d$ with times $(a, a + \text{dwell})$ and metadata (travel~$\tau$, route geometry, mode).
      \STATE Advance cursor: $t = a + \text{dwell}$ (compact), or $t = a + \text{dwell} + \text{extra\_gap}$ (sparse).
    \ELSE
      \IF{$p$ is fixed to day $d$ via preferredDay}
        \STATE Push stop $p$ to overflow list (fixed-day stop could not fit within day $d$).
        \STATE Remove $p$ from the unscheduled set so it is not considered on later days.
      \ENDIF
      \STATE Spill stop $p$ to next day.
      \STATE If $p$ is unassigned, it remains eligible on subsequent days until scheduled or pushed to overflow at the end of the date range.
    \ENDIF
  \ENDWHILE
\ENDFOR
\STATE Output: list of days with scheduled stops \& overflow list of unscheduled places (if any exceed date range).
\end{algorithmic}
\end{algorithm}
